alter session set NLS_DATE_FORMAT='DD/MM/YYYY'; 

CREATE TABLE PERSON
(
  Person_id      NUMBER(4) PRIMARY KEY,
  Name		     CHAR(30), 
  Address 	 	 VARCHAR2(50), 
  Phone_Num      VARCHAR2(15),
  DOB            DATE, 
  Postcode       NUMBER(5), 
  State          CHAR(15), 
  Gender         CHAR(1)
 );

CREATE TABLE BRANCH
(
  Branch_id      VARCHAR2(4) PRIMARY KEY, 
  Address 	     VARCHAR2(50),
  Type           CHAR(10), 
  Postcode       NUMBER(5)
);

CREATE TABLE MODEL
(
  Model_no       VARCHAR2(4) PRIMARY KEY,
  Model_name     VARCHAR2(30), 
  Model_storage  NUMBER(5),
  Model_RAM      NUMBER(5),
  Model_Type     VARCHAR2(20),
  Model_Price    NUMBER(10,2)
);

CREATE TABLE SUPPLIER
(
  Supplier_id    VARCHAR2(5) PRIMARY KEY,
  Phone_Num      VARCHAR2(15),
  Address 	     VARCHAR2(50), 
  Postcode       NUMBER(5)
);

CREATE TABLE COURIER
(
  Courier_id            VARCHAR(5) PRIMARY KEY,
  Name                  CHAR(10), 
  Phone_No              VARCHAR2(15),
  Price_per_delivery    NUMBER(3)
);

CREATE TABLE STAFF
(
  Staff_id      NUMBER(4) PRIMARY KEY, 
  Position		VARCHAR2(15),
  Salary        NUMBER(4),
  Branch_id     VARCHAR2(4),
  FOREIGN KEY (Branch_id) references BRANCH(Branch_id)
);

CREATE TABLE PRODUCT
(
  Product_id     VARCHAR(4) PRIMARY KEY,
  Model_no       VARCHAR2(10),
  Description    VARCHAR2(30),
  FOREIGN KEY (Model_no) references MODEL(Model_no)
);

CREATE TABLE STOCK
(
  Stock_id       VARCHAR2(8) PRIMARY KEY,
  Supplier_id    VARCHAR2(5), 
  Stock_date     DATE, 
  Branch_id      VARCHAR2(4),
  FOREIGN KEY (Branch_id) references BRANCH(Branch_id)
);

CREATE TABLE PRODUCT_STOCK
(
  CONSTRAINT PK_PRODUCT_STOCK PRIMARY KEY (Product_id, Stock_id),
  Product_id     VARCHAR(4),
  Stock_id       VARCHAR(8),
  Quantity       NUMBER(5),
  FOREIGN KEY (Product_id) references PRODUCT(Product_id),
  FOREIGN KEY (Stock_id) references STOCK(Stock_id)
);


CREATE TABLE SALES
(
  Sales_id       VARCHAR2(7) PRIMARY KEY, 
  Sales_date     DATE,
  Staff_ID		 NUMBER(4),
  Cust_ID        NUMBER(4),
  Mode_sales     CHAR(10), 
  FOREIGN KEY (Staff_id) references STAFF(Staff_id),
  FOREIGN KEY (Cust_id) references PERSON(Person_id)
);

CREATE TABLE SALES_PRODUCT
(
  CONSTRAINT PK_SALES_PRODUCT PRIMARY KEY (Sales_id, Product_id),
  Sales_id     VARCHAR2(7),
  Product_id   VARCHAR2(4),
  Quantity     NUMBER(5),
  FOREIGN KEY (Sales_id) references SALES(Sales_id),
  FOREIGN KEY (Product_id) references PRODUCT(Product_id)
);

CREATE TABLE DELIVERY
(
  Delivery_id    VARCHAR2(5) PRIMARY KEY,
  Sales_id       VARCHAR2(7),
  Courier_id     VARCHAR2(5),
  Tracking_no    VARCHAR2(20),
  Dep_date       DATE, 
  Arrival_date   DATE, 
  Arrival_time   NUMBER(4),
  FOREIGN KEY (Sales_id) references SALES(Sales_id),
  FOREIGN KEY (Courier_id) references COURIER(Courier_id)
);
